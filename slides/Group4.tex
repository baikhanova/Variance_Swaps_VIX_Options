% ── Slide 4: Heston Model ──
\begin{frame}{W2 -- Mathematical Derivations (Model Framework)}

\begin{columns}[T]
\column{0.48\textwidth}
\textbf{Heston model \& variance swap}
\begin{itemize}
  \item Heston SDE:\; $dv_t = \kappa(\theta - v_t)\,dt + \sigma\sqrt{v_t}\,dW_t$
  \item Discounted stock is $\mathbb{Q}$-martingale:\; $\mathbb{E}^{\mathbb{Q}}[e^{-rT}S_T] = S_0$
  \item Variance swap fair strike:
  \[
    K_{\mathrm{var}} = \theta + \frac{(v_0 - \theta)(1 - e^{-\kappa T})}{\kappa T}
  \]
  \item Derivation via It\^o's lemma on $\ln S_t$ and replication portfolio
\end{itemize}

\column{0.48\textwidth}
\textbf{VIX affine formula}
\begin{itemize}
  \item $\mathrm{VIX}_t^2$ affine in $v_t$ under Heston:
  \[
    \mathrm{VIX}_t^2 = \frac{1}{\Delta}\,\mathbb{E}_t^{\mathbb{Q}}\!\Bigl[\int_t^{t+\Delta}\! v_s\,ds\Bigr] = \theta + (v_t - \theta)\,B(\kappa,\Delta)
  \]
  \item $B = \frac{1 - e^{-\kappa\Delta}}{\kappa\Delta}$ \;(mean-reversion weight)
  \item $\mathrm{VIX}_t = 100\sqrt{A + B\cdot v_t}$,\; $A = \theta(1-B)$
\end{itemize}
\end{columns}

\vspace{6pt}
\begin{block}{Joint Calibration Problem --- Why Heston Fails?}
\begin{itemize}
  \item One-factor: $v_t$ drives both SPX implied vol and VIX term structure simultaneously
  \item Fitting SPX surface requires specific $(\kappa, \sigma_v)$; VIX curve then deviates from market
  \item Resolution: two-factor CIR ($v_t = v_t^{(1)} + v_t^{(2)}$) or rough Heston
\end{itemize}
\end{block}
\end{frame}

% ── Slide 5: CIR CF Derivation ──
\begin{frame}{W2 -- CIR Characteristic Function Derivation}

\textbf{Goal:} derive $\varphi(u;\tau) = \mathbb{E}^{\mathbb{Q}}[\exp(i u\cdot v_T) \mid v_0]$ for terminal CIR variance.

\medskip
\textbf{Step 1: Feynman--Kac.}\;
Set $F(v,t)=\mathbb{E}[e^{iuv_T}\mid v_t=v]$. PDE:
\[
  \frac{\partial F}{\partial t} + \kappa(\theta-v)\frac{\partial F}{\partial v}
  + \tfrac{1}{2}\sigma^2 v\,\frac{\partial^2 F}{\partial v^2} = 0,
  \qquad F(v,T)=e^{iuv}.
\]

\textbf{Step 2: Ansatz.}\;
Try $F = \exp\!\bigl(A(\tau)+B(\tau)\cdot v\bigr)$, $\tau=T-t$.
Substituting $\Rightarrow$ Riccati ODE for $B(\tau)$ and linear ODE for $A(\tau)$.

\medskip
\textbf{Step 3: Riccati solution.}
\[
  \gamma = \sqrt{\kappa^2 - 2\sigma^2 iu},\quad
  D(\tau)=(\kappa+\gamma)+(\gamma-\kappa)e^{-\gamma\tau}
\]
\[
  B(\tau) = \frac{2iu(1-e^{-\gamma\tau})}{D(\tau)},\qquad
  A(\tau) = \frac{\kappa\theta}{\sigma^2}\Bigl[(\kappa-\gamma)\tau - 2\ln\frac{D(\tau)}{2\gamma}\Bigr]
\]

\textbf{Step 4: Final CF.}\;
$\varphi(u;\tau)=\exp\!\bigl(A(\tau)+B(\tau)\cdot v_0\bigr)$.\\
Boundary check: $\varphi(0;\tau)=1$ \;\checkmark\qquad $|\varphi(u;\tau)|\le 1$ for real $u$ \;\checkmark
\end{frame}

% ── Slide 6: Monte Carlo ──
\begin{frame}{W3 -- Monte Carlo Engine \;(\texttt{variance\_mc.py})}

\begin{description}
  \item[Heston path simulation]
    Full-truncation Euler scheme; correlated BM for $(S_t, v_t)$;
    variance floored at 0 to avoid negative values.

  \item[Realised variance]
    $\mathrm{RV} = \sum ({\ln S_{t_i}/S_{t_{i-1}}})^2 / T$ \;(annualised);
    compared to analytical $K_{\mathrm{var}}$ --- error ${<}\,1\%$ at 10\,000 paths.

  \item[VIX call by MC]
    Simulate $\mathrm{VIX}_T = 100\sqrt{A+B\cdot v_T}$ at option maturity;
    payoff $= \max(\mathrm{VIX}_T - K,\,0)$; discounted expectation.

  \item[Convergence study]
    Path counts $500 \to 50\,000$;
    MC error $\sim 1/\sqrt{N}$ as expected; CI width plotted.

  \item[VRP simulation]
    $\mathrm{VRP}_{\mathrm{sim}} = v_0 - \mathrm{RV}_{0,T}$ per path; distribution is right-skewed
    with positive mean $\approx 2$--$3$ vol pts confirming the VRP.
\end{description}
\end{frame}

% ── Slide 7: COS Engine ──
\begin{frame}{W4 -- Fourier Engineer: COS Pricing Engine \;(\texttt{vix\_cos.py})}

\textbf{Core idea:} express the expected payoff as a Fourier-cosine series using the known characteristic function of terminal variance.

\begin{columns}[T]
\column{0.35\textwidth}
\textbf{COS density recovery}
\[
  f(v) \approx \frac{2}{b-a}\sideset{}{'}\sum_k \mathrm{Re}[\varphi_k]\cos\!\Bigl(\frac{k\pi(v-a)}{b-a}\Bigr)
\]
Density integrates to 1 within 0.5\%.

\column{0.35\textwidth}
\textbf{Expected payoff via COS}
\[
  \mathbb{E}[g(v_T)] \approx \sideset{}{'}\sum_k \mathrm{Re}\!\bigl[\varphi_k\, e^{-ik\pi a/(b-a)}\bigr] V_k
\]
$V_k$ = payoff cosine coefficient.\\
Vectorised: $O(N\!\cdot\!M)$.

\column{0.28\textwidth}
\textbf{Pricing functions}
\begin{itemize}\small
  \item \texttt{price\_vix\_call\_cos}
  \item \texttt{price\_vix\_put\_cos}
  \item \texttt{price\_vix\_futures\_cos}
\end{itemize}
\end{columns}

\vspace{4pt}
\textbf{Truncation interval:}\; $a=0$ (CIR non-negative),\; $b = \mu + L\sigma$ ($L=8$ std widths), based on closed-form CIR moments.
\end{frame}

% ── Slide 8: COS Algorithm Step-by-Step ──
\begin{frame}{W4 -- COS Pricing Algorithm: Step by Step}
\begin{enumerate}
  \item \textbf{Determine interval $[a,b]$:}\;
    $a=0$,\; $b=\mathbb{E}[v_T]+8\sqrt{\mathrm{Var}[v_T]}$ \;(CIR closed-form moments).

  \item \textbf{Compute characteristic function:}
    \[
      \varphi_k = \mathrm{Re}\!\Bigl[\varphi\!\bigl(k\pi/(b\!-\!a)\bigr)\cdot e^{-ik\pi a/(b-a)}\Bigr],
      \quad k=0,\dots,N\!-\!1.
    \]

  \item \textbf{Compute payoff coefficients $V_k$:}
    \[
      V_k = \frac{2}{b-a}\int_a^b g(v)\cos\!\Bigl(\frac{k\pi(v-a)}{b-a}\Bigr)dv,
      \quad g(v)=\max\!\bigl(100\sqrt{A+Bv}-K,\,0\bigr).
    \]

  \item \textbf{COS summation:}
    $\displaystyle\mathbb{E}[g(v_T)] \approx \sideset{}{'}\sum_{k=0}^{N-1} \varphi_k \cdot V_k$
    \;(prime: $k\!=\!0$ term $\times\tfrac{1}{2}$).\\
    Vectorised via $(N\!\times\!M)$ matrix --- no loop.

  \item \textbf{Discount:}\; $\mathrm{Price} = e^{-rT}\cdot\mathbb{E}[g(v_T)]$.
\end{enumerate}
\end{frame}

% ── Slide 14: COS vs MC Comparison ──
\begin{frame}{W4 -- Method Comparison: COS vs.\ Monte Carlo}

Both methods price the same VIX call. COS uses the analytical CF; MC uses simulated paths.

\begin{columns}[T]
\column{0.45\textwidth}
\begin{block}{COS Method}
\begin{itemize}\small
  \item Characteristic function + Fourier series
  \item $O(N\cdot M)$ vectorised --- runs in milliseconds
  \item $N=64$ gives 4-decimal accuracy
  \item No MC variance --- deterministic
  \item Truncation error well-controlled by \texttt{std\_width}
  \item Used for: call, put, futures, density, term structure
\end{itemize}
\end{block}

\column{0.45\textwidth}
\begin{block}{Monte Carlo Method}
\begin{itemize}\small
  \item Simulate 10\,000+ Heston/CIR paths
  \item $O(\text{paths}\cdot\text{steps})$ --- slower but intuitive
  \item Confidence interval explicit via CLT
  \item MC standard error $\approx \sigma/\sqrt{N} \approx 0.10$ for $N\!=\!10$K
  \item Validates COS within MC noise band
  \item Used for: VRP distribution, P\&L, model check
\end{itemize}
\end{block}
\end{columns}

\vspace{6pt}
\centering
COS price $\approx$ MC price within MC standard error \;(abs.\ diff ${<}\,0.10$ VIX pts).\\[2pt]
Both methods converge to the same expectation --- cross-validation confirms engine correctness.
\end{frame}
